# 0 for facing north
0 xE** -> N 0
0 *x** -> E 1
0 NExx -> W 2
0 NEWx -> S 3

# 1 for facing east
1 xExS -> N 0
1 *xxS -> E 1
1 NExS -> W 2
1 **xx -> S 3

# 2 for facing west
2 Nxx* -> W 2
2 x*** -> N 0
2 NxWS -> E 1
2 NxWx -> S 3

# 3 for facing south
3 **Wx -> S 3
3 **x* -> W 2
3 xxWS -> E 1
3 xEWS -> N 0
